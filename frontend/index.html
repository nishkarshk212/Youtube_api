API Endpoints

            
            

                
                

                    

                        
Search Songs

                        GET
                    

                    /api/search?song={song_name}
                    
Search for songs on YouTube. Returns a list of matching songs with metadata.

                    

                        
Parameters:

                        
    
                            
  • song (required) - Song name to search for
    • 
                        

                    

                    

                        
Response:

                        
[
  {
    "video_id": "dQw4w9WgXcQ",
    "title": "Never Gonna Give You Up",
    "duration": 212,
    "thumbnail": "https://i.ytimg.com/vi/dQw4w9WgXcQ/hqdefault.jpg",
    "uploader": "Rick Astley",
    "view_count": 1400000000,
    "upload_date": "20091025",
    "url": "https://www.youtube.com/watch?v=dQw4w9WgXcQ"
  }
]

                    

                


                
                

                    

                        
Play Song

                        GET
                    

                    /api/play?song={song_name}
                    
Get direct audio information for a song. Returns the first search result.

                


                
                

                    

                        
Video Details

                        GET
                    

                    /api/video?id={video_id}
                    
Get detailed video information including description, tags, and categories.

                


                
                

                    

                        
Audio Stream

                        GET
                    

                    /api/audio?id={video_id}
                    
Get downloadable audio stream URL with quality information.

                


                
                

                    

                        
Lyrics

                        GET
                    

                    /api/lyrics?song={song_name}
                    
Get lyrics for a song from video captions or auto-generated captions.

                


                
                

                    

                        
Playlist

                        GET
                    

                    /api/playlist?id={playlist_id}
                    
Get playlist information and all tracks with metadata.

                


                
                

                    

                        
Trending

                        GET
                    

                    /api/trending?category={category}
                    
Get trending videos in a specific category (default: music).

                


                
                

                    

                        
Related Videos

                        GET
                    

                    /api/related?id={video_id}
                    
Get related videos for a given video.

                


                
                

                    

                        
Health Check

                        GET
                    

                    /api/health
                    
Check API status, cache status, and extractor version.

                

            

        

    


    
    

        

            
Usage Examples

            
            

                
                

                    

                        Python
                    

                    
import requests

API_BASE = "http://your-api-url.com/api"
API_KEY = "your-api-key"

headers = {"X-API-Key": API_KEY}

# Search for a song
response = requests.get(
    f"{API_BASE}/search",
    params={"song": "Never Gonna Give You Up"},
    headers=headers
)
results = response.json()

# Get audio stream
video_id = results[0]["video_id"]
audio = requests.get(
    f"{API_BASE}/audio",
    params={"id": video_id},
    headers=headers
).json()

print(f"Stream URL: {audio['url']}")

                


                
                

                    

                        JavaScript
                    

                    
const API_BASE = "http://your-api-url.com/api";
const API_KEY = "your-api-key";

const headers = {
    "X-API-Key": API_KEY
};

// Search for a song
async function searchSong(songName) {
    const response = await fetch(
        `${API_BASE}/search?song=${encodeURIComponent(songName)}`,
        { headers }
    );
    return await response.json();
}

// Get audio stream
async function getAudioStream(videoId) {
    const response = await fetch(
        `${API_BASE}/audio?id=${videoId}`,
        { headers }
    );
    return await response.json();
}

// Usage
const results = await searchSong("Never Gonna Give You Up");
const audio = await getAudioStream(results[0].video_id);
console.log("Stream URL:", audio.url);

                


                
                

                    

                        Pyrogram (Telegram Bot)
                    

                    
from pyrogram import Client, filters
import requests

API_BASE = "http://your-api-url.com/api"
API_KEY = "your-api-key"

@app.on_message(filters.command("play"))
async def play_song(client, message):
    song_name = " ".join(message.command[1:])
    
    # Search for song
    response = requests.get(
        f"{API_BASE}/search",
        params={"song": song_name},
        headers={"X-API-Key": API_KEY}
    )
    results = response.json()
    
    if not results:
        await message.reply("Song not found!")
        return
    
    video = results[0]
    
    # Get audio stream
    audio = requests.get(
        f"{API_BASE}/audio",
        params={"id": video["video_id"]},
        headers={"X-API-Key": API_KEY}
    ).json()
    
    # Send audio to chat
    await message.reply_audio(
        audio["url"],
        title=video["title"],
        duration=video["duration"],
        thumb=video["thumbnail"]
    )

                


                
                

                    

                        Telethon (Telegram Bot)
                    

                    
from telethon import TelegramClient, events
import requests

API_BASE = "http://your-api-url.com/api"
API_KEY = "your-api-key"

@client.on(events.NewMessage(pattern='/play'))
async def handler(event):
    song_name = event.message.message.split('/play ')[1]
    
    # Search for song
    response = requests.get(
        f"{API_BASE}/search",
        params={"song": song_name},
        headers={"X-API-Key": API_KEY}
    )
    results = response.json()
    
    if not results:
        await event.reply("Song not found!")
        return
    
    video = results[0]
    
    # Get audio stream
    audio = requests.get(
        f"{API_BASE}/audio",
        params={"id": video["video_id"]},
        headers={"X-API-Key": API_KEY}
    ).json()
    
    # Send audio to chat
    await event.reply(
        f"🎵 {video['title']}\n"
        f"👤 {video['uploader']}\n"
        f"⏱️ {video['duration']}s\n"
        f"🔗 {audio['url']}"
    )

Developer Documentation

            
            

                

                    

                        Authentication
                    

                    
All API requests require an API key sent in the X-API-Key header.

                    
curl -H "X-API-Key: your-api-key" \
  "http://your-api-url.com/api/search?song=example"

                


                

                    

                        Rate Limiting
                    

                    
API requests are rate limited based on your plan. Exceeding limits will return a 429 status code.

                    
{
  "error": "Rate limit exceeded",
  "detail": "Too many requests"
}

                


                

                    

                        Caching
                    

                    
Responses are cached for 1 hour by default. Stream URLs are cached for 30 minutes.

                    
Cache-Control: max-age=3600
X-Cache: HIT

                


                

                    

                        Error Handling
                    

                    
All errors return a JSON response with error details and status code.

                    
{
  "error": "Video not found",
  "detail": "The requested video does not exist",
  "status_code": 404
}
